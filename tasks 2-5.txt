2.1.
select FirstName, LastName, count(*) AS DuplicateCount
from Employees
group by FirstName, LastName
having count(*) > 1;

2.2
select * into #UniqueRows
from Employees as E1
where CreatedAt = (select max(CreatedAt)
                          from Employees as E2
                          where E1.FirstName = E2.FirstName
                            and E1.LastName = E2.LastName);

delete from Employees
where Identifier not in (select Identifier from #UniqueRows);


3.1.	
select *
from EMPLOYEES e
left join DEPARTMENTS d ON e.DEPARTMENT_ID = d.DEPARTMENT_ID;

3.2.
select d.DEPARTMENT_NAME, avg(e.SALARY) AS AVERAGE_SALARY
from EMPLOYEES e
left join DEPARTMENTS d ON e.DEPARTMENT_ID = d.DEPARTMENT_ID
group by d.DEPARTMENT_NAME;


4. 
select DEPT_ID, min(SALARY) as MIN_SALARY, max(SALARY) as MAX_SALARY
from EMPLOYEES
group by DEPT_ID
having min(SALARY) < 5000 and max(SALARY) > 15000;

5.2.
- Що станеться і чому:
SQL-синтаксис для операції DELETE не дозволяє вказувати конкретні колонки для видалення, тому виникне помилка.
